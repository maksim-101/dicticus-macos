    ```swift
    import SwiftUI
    import WhisperKit

    /// Errors thrown by TranscriptionService during the record-transcribe cycle.
    enum TranscriptionError: Error, Sendable {
        /// Recording was shorter than minimumDurationSeconds (D-06)
        case tooShort
        /// Energy-based VAD detected no voice activity (D-05)
        case silenceOnly
        /// WhisperKit returned no transcription results
        case noResult
        /// WhisperKit instance not available (model not warmed up)
        case modelNotReady
    }

    /// Core ASR pipeline: record audio via WhisperKit AudioProcessor, apply VAD,
    /// transcribe via Whisper large-v3-turbo, auto-detect language (de/en).
    ///
    /// Consumes ModelWarmupService.whisperKitInstance directly (D-10, D-13).
    /// Does NOT create its own WhisperKit instance.
    ///
    /// Phase 3 will call startRecording() on hotkey press and
    /// stopRecordingAndTranscribe() on hotkey release.
    @MainActor
    class TranscriptionService: ObservableObject {
        /// Pipeline state machine
        enum State: Sendable {
            case idle
            case recording
            case transcribing
        }

        @Published var state: State = .idle
        @Published var lastResult: DicticusTranscriptionResult?
        @Published var error: String?

        private let whisperKit: WhisperKit

        /// Energy threshold for silence pre-filter (D-05, D-07).
        /// Configurable internally for development tuning, not user-facing in v1.
        /// Starting value 0.3 matches WhisperAX example app.
        var silenceThreshold: Float = 0.3

        /// Minimum recording duration in seconds (D-06).
        /// Sub-0.3s clips are noise, not speech.
        let minimumDurationSeconds: Float = 0.3

        /// Initialize with a warm WhisperKit instance from ModelWarmupService (D-10, D-13).
        /// - Parameter whisperKit: Initialized WhisperKit from ModelWarmupService.whisperKitInstance
        init(whisperKit: WhisperKit) {
            self.whisperKit = whisperKit
        }

        /// Start recording via WhisperKit's AudioProcessor (D-01, D-02, D-03).
        /// AudioProcessor internally installs an AVAudioEngine tap at 16kHz mono
        /// and accumulates buffers -- no manual AVAudioEngine code needed.
        func startRecording() throws {
            guard state == .idle else { return }
            // WhisperKit AudioProcessor handles: tap installation, 16kHz resampling,
            // buffer accumulation (satisfies D-01, D-02, D-03 internally)
            try whisperKit.audioProcessor.startRecordingLive { _ in
                // Callback runs on audio thread per buffer.
                // No @MainActor work here (Pitfall 6: Swift 6 Sendable compliance).
                // Recording monitoring handled by audioProcessor.relativeEnergy.
            }
            state = .recording
        }

        /// Stop recording and run the full transcription pipeline.
        ///
        /// Pipeline steps:
        /// 1. Stop AudioProcessor recording (Pitfall 3: stop before reading samples)
        /// 2. Check minimum duration (D-06: >= 0.3s)
        /// 3. Energy-based VAD pre-filter (D-05: skip if silence only)
        /// 4. Transcribe with auto language detection (D-08, D-11, D-12)
        /// 5. Restrict detected language to de/en (D-11)
        /// 6. Build DicticusTranscriptionResult (Pitfall 4: copy values immediately)
        ///
        /// - Returns: DicticusTranscriptionResult with text, language, confidence
        /// - Throws: TranscriptionError for pipeline failures
        func stopRecordingAndTranscribe() async throws -> DicticusTranscriptionResult {
            whisperKit.audioProcessor.stopRecording()
            state = .transcribing

            // Copy audio samples after stop (Pitfall 3: safe to read after stopRecording)
            let audioSamples = Array(whisperKit.audioProcessor.audioSamples)
            let durationSeconds = Float(audioSamples.count) / Float(WhisperKit.sampleRate)

            // D-06: Discard sub-0.3s clips
            guard durationSeconds >= minimumDurationSeconds else {
                state = .idle
                throw TranscriptionError.tooShort
            }

            // D-05: Energy-based pre-filter before inference
            let energy = whisperKit.audioProcessor.relativeEnergy
            let hasVoice = AudioProcessor.isVoiceDetected(
                in: energy,
                nextBufferInSeconds: durationSeconds,
                silenceThreshold: silenceThreshold
            )
            guard hasVoice else {
                state = .idle
                throw TranscriptionError.silenceOnly
            }

            // D-08, D-11, D-12: Transcribe with language auto-detection
            // temperature 0.0 = greedy decoding for fastest inference (TRNS-02)
            // language nil = auto-detect (D-12)
            // chunkingStrategy .vad = WhisperKit internal VAD chunking (D-04)
            // withoutTimestamps true = no timestamps needed for dictation
            let options = DecodingOptions(
                verbose: false,
                task: .transcribe,
                language: nil,
                temperature: 0.0,
                temperatureFallbackCount: 3,
                sampleLength: 224,
                usePrefillPrompt: true,
                usePrefillCache: true,
                skipSpecialTokens: true,
                withoutTimestamps: true,
                suppressBlank: true,
                noSpeechThreshold: 0.6,
                chunkingStrategy: .vad
            )

            let results = try await whisperKit.transcribe(
                audioArray: audioSamples,
                decodeOptions: options
            )

            guard let result = results.first else {
                state = .idle
                throw TranscriptionError.noResult
            }

            // D-11: Restrict detected language to de/en
            let detectedLanguage = restrictLanguage(result.language)

            // Pitfall 4: TranscriptionResult is a class since v0.15.0 --
            // copy values into our value-type struct immediately
            let transcriptionResult = DicticusTranscriptionResult(
                text: result.text.trimmingCharacters(in: .whitespacesAndNewlines),
                language: detectedLanguage,
                confidence: 1.0 - (result.segments.first?.noSpeechProb ?? 0.0)
            )

            lastResult = transcriptionResult
            state = .idle
            return transcriptionResult
        }

        /// Restrict detected language to the allowed set {de, en} (D-11).
        /// If Whisper detects a language outside this set, default to English.
        func restrictLanguage(_ detected: String) -> String {
            let allowed: Set<String> = ["de", "en"]
            return allowed.contains(detected) ? detected : "en"
        }
    }
    ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Microphone -> AudioProcessor | Raw audio from hardware enters the app process |
| AudioProcessor -> TranscriptionService | Audio samples cross from WhisperKit's recording thread to main actor |
| TranscriptionService -> WhisperKit inference | Audio array passed to ML model for transcription |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-01 | Information Disclosure | TranscriptionService | mitigate | All audio processing is local -- no network calls in TranscriptionService. Audio samples exist only in memory during the record-transcribe cycle and are released when audioProcessor resets. Project constraint enforces this. |
| T-02-02 | Denial of Service | TranscriptionService.stopRecordingAndTranscribe | mitigate | Duration guard (>= 0.3s) prevents processing trivially small buffers. Push-to-talk bounded by user action prevents unbounded recording. Energy pre-filter skips inference on silence. |
| T-02-03 | Tampering | AudioProcessor.audioSamples | accept | Audio samples in memory could theoretically be tampered by a process with same-user privileges, but this is a single-user desktop app with no inter-process audio sharing. Hardened Runtime enabled. |
| T-02-04 | Information Disclosure | DicticusTranscriptionResult | mitigate | Result struct contains only transcribed text, language code, and confidence score. No raw audio stored in the result. lastResult is replaced on each new transcription -- no history accumulation. |
</threat_model>
