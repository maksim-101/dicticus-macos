    import Foundation
    import Combine
    import os.log

    /// Foreground URLSession downloader for the Gemma 4 E2B GGUF.
    /// Per D-10: inline Settings download UI consumes this class's `@Published` state/progress.
    /// Per D-11: destination is `Application Support/Dicticus/Models/gemma-4-E2B-it-Q4_K_M.gguf`.
    /// Per Q6: downloaded file is excluded from iCloud backup.
    @MainActor
    public final class IOSModelDownloadService: NSObject, ObservableObject, URLSessionDownloadDelegate {

        // MARK: - Public types

        public enum DownloadState: Equatable {
            case idle
            case downloading
            case paused
            case completed
            case failed(String)
        }

        // MARK: - Constants (mirror macOS ModelDownloadService per D-11)

        public static let modelURL = URL(string: "https://huggingface.co/unsloth/gemma-4-E2B-it-GGUF/resolve/main/gemma-4-E2B-it-Q4_K_M.gguf")!
        public static let modelFileName = "gemma-4-E2B-it-Q4_K_M.gguf"

        public static func modelPath() -> URL {
            let appSupport = FileManager.default.urls(
                for: .applicationSupportDirectory,
                in: .userDomainMask
            ).first!
            return appSupport
                .appendingPathComponent("Dicticus")
                .appendingPathComponent("Models")
                .appendingPathComponent(modelFileName)
        }

        public static func isModelCached() -> Bool {
            FileManager.default.fileExists(atPath: modelPath().path)
        }

        // MARK: - Published state

        @Published public private(set) var state: DownloadState = .idle
        @Published public private(set) var progress: Double = 0
        @Published public private(set) var bytesPerSec: Double = 0

        // MARK: - Internals

        private let log = Logger(subsystem: "com.dicticus", category: "download")
        private var session: URLSession!
        private var task: URLSessionDownloadTask?
        private var resumeData: Data?
        private var lastBytes: Int64 = 0
        private var lastSampleAt: Date = .distantPast
        private var completionContinuation: CheckedContinuation<Void, Error>?

        // MARK: - Init (test seam)

        public init(sessionConfiguration: URLSessionConfiguration = .default) {
            super.init()
            sessionConfiguration.waitsForConnectivity = true
            // Queue must be non-nil and must NOT be main — URLSession docs warn that main-queue delegate callbacks can deadlock with sync reads. Use a dedicated queue and hop to @MainActor for @Published mutations.
            let queue = OperationQueue()
            queue.maxConcurrentOperationCount = 1
            queue.name = "com.dicticus.IOSModelDownloadService"
            self.session = URLSession(configuration: sessionConfiguration, delegate: self, delegateQueue: queue)
        }

        // MARK: - Public API

        public func start() {
            // Ensure parent directory exists.
            let parent = Self.modelPath().deletingLastPathComponent()
            try? FileManager.default.createDirectory(at: parent, withIntermediateDirectories: true)

            if let resumeData {
                task = session.downloadTask(withResumeData: resumeData)
                self.resumeData = nil
            } else {
                task = session.downloadTask(with: Self.modelURL)
            }
            state = .downloading
            lastBytes = 0
            lastSampleAt = Date()
            task?.resume()
        }

        public func pause() {
            guard let task else { return }
            task.cancel(byProducingResumeData: { [weak self] data in
                Task { @MainActor in
                    self?.resumeData = data
                    self?.state = .paused
                }
            })
            self.task = nil
        }

        public func resume() { start() }

        /// Test helper: start and await completion or failure.
        public func startAndWaitForCompletion() async throws {
            try await withCheckedThrowingContinuation { (cont: CheckedContinuation<Void, Error>) in
                self.completionContinuation = cont
                self.start()
            }
        }

        /// Test helper: await whatever in-flight download completes.
        public func waitForCompletion() async {
            while state == .downloading || state == .paused {
                try? await Task.sleep(nanoseconds: 50_000_000)
            }
        }

        // MARK: - URLSessionDownloadDelegate

        public nonisolated func urlSession(
            _ session: URLSession,
            downloadTask: URLSessionDownloadTask,
            didWriteData bytesWritten: Int64,
            totalBytesWritten: Int64,
            totalBytesExpectedToWrite: Int64
        ) {
            // Snapshot values, hop to MainActor for publishing.
            let written = totalBytesWritten
            let expected = totalBytesExpectedToWrite
            Task { @MainActor in
                guard expected > 0 else { return }
                self.progress = Double(written) / Double(expected)
                let now = Date()
                let elapsed = now.timeIntervalSince(self.lastSampleAt)
                if elapsed > 0.5 {
                    let delta = Double(written - self.lastBytes)
                    self.bytesPerSec = delta / elapsed
                    self.lastBytes = written
                    self.lastSampleAt = now
                }
            }
        }

        public nonisolated func urlSession(
            _ session: URLSession,
            downloadTask: URLSessionDownloadTask,
            didFinishDownloadingTo location: URL
        ) {
            // Called on delegate queue. Move file synchronously while URL is valid,
            // then hop to MainActor for state publish.
            let dest = Self.modelPath()
            let parent = dest.deletingLastPathComponent()
            var moveError: Error?
            do {
                try FileManager.default.createDirectory(at: parent, withIntermediateDirectories: true)
                try? FileManager.default.removeItem(at: dest)
                try FileManager.default.moveItem(at: location, to: dest)

                // Q6: exclude from iCloud backup.
                var urlRef = dest
                var values = URLResourceValues()
                values.isExcludedFromBackup = true
                try urlRef.setResourceValues(values)
            } catch {
                moveError = error
            }

            Task { @MainActor in
                if let moveError {
                    self.state = .failed("Failed to save model: \(moveError.localizedDescription)")
                    self.completionContinuation?.resume(throwing: moveError)
                } else {
                    self.progress = 1.0
                    self.state = .completed
                    self.completionContinuation?.resume()
                }
                self.completionContinuation = nil
                self.task = nil
            }
        }

        public nonisolated func urlSession(
            _ session: URLSession,
            task: URLSessionTask,
            didCompleteWithError error: Error?
        ) {
            guard let error else { return }
            let nsError = error as NSError
            // If the user paused, don't treat cancel as failure.
            if nsError.domain == NSURLErrorDomain && nsError.code == NSURLErrorCancelled {
                return
            }
            Task { @MainActor in
                self.state = .failed(error.localizedDescription)
                self.completionContinuation?.resume(throwing: error)
                self.completionContinuation = nil
            }
        }
    }

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| HuggingFace CDN ↔ iOS device | Downloaded GGUF (~3.1 GB) crosses network. Source is `huggingface.co/unsloth` — ungated, public, CDN-hosted. |
| Downloaded file ↔ llama.cpp loader | GGUF format parsed by native llama.cpp code after download. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-19-03-01 | T (Tampering) | GGUF file in transit | mitigate | Download over HTTPS (URL scheme `https://`); URLSession validates server certificate by default. HF CDN uses Fastly/Cloudflare with valid certs. No user-overridable pinning needed for a public model. |
| T-19-03-02 | T (Tampering) | GGUF file at rest | accept | File stored in `Application Support/Dicticus/Models/` (app-sandboxed). Only code with the app's sandbox identity can read/write. No public checksum validation added in v2.1 — deferred. |
| T-19-03-03 | I (Info Disclosure) | GGUF in iCloud Backup | mitigate | `isExcludedFromBackup = true` set via URLResourceValues right after move (Q6). Prevents 3 GB bloat in user iCloud backups. |
| T-19-03-04 | D (Denial of Service) | Download failure mid-flight | mitigate | Pause/resume via `cancel(byProducingResumeData:)` + Range request; `waitsForConnectivity = true` on session lets iOS handle transient loss. User retries from UI if `state == .failed`. |
| T-19-03-05 | S (Spoofing) | Redirect attack to malicious mirror | accept | HF CDN redirects are handled by URLSession's default delegate behavior. Since the model is Q4_K_M GGUF and llama.cpp validates the magic bytes, a malicious file fails load (goes to `state = .failed` via D-26 cleanup fallback, no user harm). |
</threat_model>
