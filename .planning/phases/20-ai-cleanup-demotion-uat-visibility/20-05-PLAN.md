UserDefaults key (NEW, defined as a Shared/ constant for both platforms):
```swift
public enum CleanupCopyMode: String {
    case raw, polished
    public static let userDefaultsKey = "cleanupCopyMode"
    public static var current: CleanupCopyMode {
        get { CleanupCopyMode(rawValue: UserDefaults.standard.string(forKey: userDefaultsKey) ?? "") ?? .raw }
        set { UserDefaults.standard.set(newValue.rawValue, forKey: userDefaultsKey) }
    }
}
```
